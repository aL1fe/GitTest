//Операции отмены
git reset HEAD <имя_файла> <имя_файла> <имя_файла> //отмена индексации файла
 
git reset --hard HEAD //отменить изменения во всех файлах включая проиндексированные (загрузить файлы из последнего коммита)
git reset --hard f2e333b35a5987ed03ce20fc2dcec002572bf3f5 //резетнуть ветку до нужного хеша (загрузить файлы из коммита суказанным хешом)

git checkout <имя_файла> //отменить изменения в отдельном файле даже если он проиндексирован

git checkout-index <имя_файла> //вернуть удаленный файл из индекса
git checkout-index -a //вернуть все файлы из индекса

git restore <имя_файла> //discard изменения в worktree
git restore '*.*' //discard изменения в worktree все файлы
git restore --staged <имя_файла> //discard изменения в index
---------------------------
git diff //сравнить worktree с index
git diff --staged //сравнить index с последним коммитом
